            System.out.println(count);

        }

        // While loop loops 20 times while returning the count of elements.
        // boolean statement determines is count value is greater or equal to 20
        // If statement is true while loop will loop 20 times.
        // while loop is a control flow statement
        while(count <= 20) ;

        while(true) {

        }

    }

}
